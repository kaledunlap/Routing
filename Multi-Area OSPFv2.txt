~Configuring Multi-area OSPFv2~

Basic config:
R1: 
enable
config t
hostname R1
no ip domain-lookup
banner motd #Unauthorized Access is Prohibited#
enable password cyber
line con 0
password security
logging synchronous
line vty 0 5
password security
exit
int Lo0
ip address 209.165.200.225 255.255.255.252
int Lo1
ip address 192.168.1.1 255.255.255.0
int Lo2
ip address 192.168.2.1 255.255.255.0
int s0/0/0
ip address 192.168.12.1 255.255.255.252
clock rate 128000
no shut
exit

(Note: No need for the "#no shutdown" command for loopback interfaces, they are enabled unless shutdown manually)

R2:
enable
config t
hostname R2
no ip domain-lookup
banner motd #Unauthorized Access is Prohibited#
enable password cyber
line con 0
password security
logging synchronous
line vty 0 5
password security
exit
int Lo6
ip address 192.168.6.1 255.255.255.0
int s0/0/0
ip address 192.168.12.2 255.255.255.252
no shut
int s0/0/1
ip address 192.168.23.1 255.255.255.252
clock rate 128000
no shut
end

(Note: DCE interfaces clock rate should be set to 128000)

R3:
enable
config t
hostname R3
no ip domain-lookup
banner motd #Unauthorized Access is Prohibited#
enable password cyber
line con 0
password security
logging synchronous
line vty 0 5
password security
exit
int Lo4
ip address 192.168.4.1 255.255.255.0
int Lo5
ip addres 192.168.5.1 255.255.255.0
int s0/0/1
ip address 192.168.23.2 255.255.255.252
no shut
end

Configure OSPF on R1
R1:
config t
router ospf 1
router-id 1.1.1.1
network 192.168.1.0 0.0.0.255 area 1
network 192.168.2.0 0.0.0.255 area 1
network 192.168.12.0 0.0.0.3 area 0
passive-interface Lo1
passive-interface Lo2
exit
ip route 0.0.0.0 0.0.0.0 Lo0
router ospf 1                
default-information originate
exit

R2:
config t
router ospf 1
router-id 2.2.2.2
network 192.168.12.0 0.0.0.3 area 0
network 192.168.6.0 0.0.0.255 area 3
network 192.168.23.0 0.0.0.3 area 3
passive-interface Lo6
exit

R3:
config t
router ospf 1
router-id 3.3.3.3
network 192.168.23.0 0.0.0.3 area 3
network 192.168.4.0 0.0.0.255 area 3
network 192.168.5.0 0.0.0.255 area 3
passive-interface Lo4
passive-interface Lo5
exit
